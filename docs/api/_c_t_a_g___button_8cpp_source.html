26void CTAG_Button::update() {
27 // Reset the one-shot event flags at the beginning of each update cycle.
28 _wasShortPressedFlag = false;
29 _wasLongPressedFlag = false;
30
31 // --- Debounce Logic ---
32 bool currentReading = digitalRead(_pin);
33
34 // If the switch changed, reset the debounce timer.
35 if (currentReading != _lastReading) {
36 _lastDebounceTime = millis();
37 }
38
39 // Check if the signal has been stable for longer than the debounce delay.
40 if ((millis() - _lastDebounceTime) > _debounceDelay) {
41 // If the stable reading is different from the button's known state, a state change has occurred.
42 if (currentReading != _buttonState) {
43 _buttonState = currentReading;
44
45 // A debounced PRESS event has just occurred.
46 if (_buttonState == LOW) {
47 _pressStartTime = millis();
48 _longPressTriggered = false; // Reset long press tracker.
49 }
50 // A debounced RELEASE event has just occurred.
51 else {
52 // If a long press wasn't triggered during the press, it must have been a short press.
53 if (!_longPressTriggered) {
54 _wasShortPressedFlag = true;
55 }
56 }
57 }
58 }
59 _lastReading = currentReading;
60
61 // --- Long Press Logic ---
62 // Check if the button is currently held down AND the long press event has not yet fired for this press.
63 if (_buttonState == LOW && !_longPressTriggered) {
64 if (millis() - _pressStartTime > _longPressDelay) {
65 _wasLongPressedFlag = true;
66 _longPressTriggered = true; // Set the flag to prevent this event from firing multiple times per press.
67 }
68 }
69}
