13#include <CTAG_LED.h>
14#include "pins_arduino.h" // Required to use standardized pin names
15
16// --- Configuration ---
17// Define which pin is used for the NeoPixel data line.
18#define PIXEL_PIN PIN_NEOPIXELS
19#define NUMPIXELS 1
20
21/**
22 * @brief Global instance of our CTAG_LED library.
23 */
24CTAG_LED leds(NUMPIXELS, PIXEL_PIN);
25
26/**
27 * @brief Variables for the non-blocking state machine.
28 */
29int colorState = 0; // Current state (0=Red, 1=Green, 2=Blue, 3=Off)
30unsigned long lastChangeTime = 0; // Timestamp of the last color change.
31const long interval = 1000; // Interval to stay on each color (in milliseconds).
32
33
34/**
35 * @brief Runs once at startup to initialize hardware.
36 */
37void setup() {
38 Serial.begin(115200);
39 Serial.println("\n--- CTAG_LED Non-Blocking Demo ---");
40
41 // Initialize the LED library.
42 // The begin() function will also turn the LED off initially.
43 leds.begin();
44
45 // Set a moderate brightness for the LED.
46 leds.setBrightness(50);
47}
48
49/**
50 * @brief Runs continuously. Uses a non-blocking timer to cycle colors.
51 */
52void loop() {
53 // Check if the desired interval has passed since the last color change.
54 if (millis() - lastChangeTime >= interval) {
55
56 // Update the timestamp for the next change.
57 lastChangeTime = millis();
58
59 // Change the LED color based on the current state.
60 switch (colorState) {
61 case 0: // State 0: Red
62 Serial.println("LED state: Red");
63 leds.setPixelColor(0, 255, 0, 0); // Set the first pixel (index 0) to Red
64 break;
65 case 1: // State 1: Green
66 Serial.println("LED state: Green");
67 leds.setPixelColor(0, 0, 255, 0); // Set the first pixel to Green
68 break;
69 case 2: // State 2: Blue
70 Serial.println("LED state: Blue");
71 leds.setPixelColor(0, 0, 0, 255); // Set the first pixel to Blue
72 break;
73 case 3: // State 3: Off
74 Serial.println("LED state: Off");
75 leds.setPixelColor(0, 0, 0, 0); // Set the first pixel to Off
76 break;
77 }
78
79 // Send the new color data to the physical LED.
80 leds.show();
81
82 // Advance to the next state for the next cycle.
83 colorState++;
84 if (colorState > 3) {
85 colorState = 0; // Reset state machine to loop back to Red.
86 }
87 }
88
89 // You can add other non-blocking code here.
90 // It will run continuously without being stopped by delays.
91}
