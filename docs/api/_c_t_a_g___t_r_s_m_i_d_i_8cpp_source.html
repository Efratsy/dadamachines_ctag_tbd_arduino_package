72void CTAG_TRSMIDI::sendPitchBend(int value, byte channel) {
73 // Convert the -8192..8191 range to the MIDI 14-bit range 0..16383.
74 unsigned int midiValue = value + 8192;
75 // Split the 14-bit value into two 7-bit data bytes.
76 byte lsb = midiValue & 0x7F; // Least Significant Byte
77 byte msb = (midiValue >> 7) & 0x7F; // Most Significant Byte
78
79 byte status_byte = 0xE0 | (channel - 1);
80 midiPort->write(status_byte);
81 midiPort->write(lsb);
82 midiPort->write(msb);
83}

109/**
110 * @brief The internal MIDI parser. A state machine that processes the MIDI stream byte by byte.
111 */
112void CTAG_TRSMIDI::parse(byte midiByte) {
113 // Ignore Real-Time messages (System & Clock).
114 if (midiByte >= 0xF8) {
115 return;
116 }
117
118 // Check if it's a new status byte (most significant bit is 1).
119 if (midiByte & 0x80) {
120 status = midiByte;
121 // Reset data bytes for the new message.
122 data1 = 0;
123 data2 = 0;
124 } else {
125 // It's a data byte.
126 if (data1 == 0) {
127 data1 = midiByte;
128
129 // Check if it's a 2-byte message (like Program Change) which is now complete.
130 byte command = status & 0xF0;
131 if (command == 0xC0 || command == 0xD0) {
132 if (handleProgramChange && command == 0xC0) {
133 handleProgramChange((status & 0x0F) + 1, data1);
134 }
135 // TODO: Channel Pressure (0xD0) could be handled here.
136 status = 0; data1 = 0; // Message processed, reset state.
137 }
138 } else {
139 data2 = midiByte;
140
141 // It's a 3-byte message, and it's now complete.
142 byte command = status & 0xF0;
143 byte channel = (status & 0x0F) + 1;
144
145 switch(command) {
146 case 0x90: // Note On
147 // Note On with velocity 0 is often treated as Note Off.
148 // This library calls the appropriate callback for clarity.
149 if (data2 > 0) {
150 if (handleNoteOn) handleNoteOn(channel, data1, data2);
151 } else {
152 if (handleNoteOff) handleNoteOff(channel, data1, data2);
153 }
154 break;
155 case 0x80: // Note Off
156 if (handleNoteOff) handleNoteOff(channel, data1, data2);
157 break;
158 case 0xB0: // Control Change
159 if (handleControlChange) handleControlChange(channel, data1, data2);
160 break;
161 case 0xE0: // Pitch Bend
162 if (handlePitchBend) {
163 // Combine the two 7-bit values into one 14-bit value.
164 int value = ((int)data2 << 7) | (int)data1;
165 // Convert back to the -8192..8191 range.
166 handlePitchBend(channel, value - 8192);
167 }
168 break;
169 // Other 3-byte messages could be added here (e.g., Polyphonic Aftertouch 0xA0).
170 }
171 // Reset data bytes for the next message. Status is kept for "Running Status".
172 data1 = 0;
173 data2 = 0;
174 }
175 }
