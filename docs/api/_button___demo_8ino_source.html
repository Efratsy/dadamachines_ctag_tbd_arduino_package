43/**
44 * @brief Runs continuously after setup().
45 */
46void loop() {
47 // This function must be called in every loop cycle!
48 // It reads the button and updates its internal state machine.
49 button.update();
50
51 // --- Check for Events ---
52 // Events are "one-shot" and only fire once when they occur.
53
54 // Event: Short Press (pressed AND released in < 0.8s)
55 if (button.wasShortPressed()) {
56 Serial.println("EVENT: Short press detected!");
57 }
58
59 // Event: Long Press (held for > 0.8s)
60 if (button.wasLongPressed()) {
61 Serial.println("EVENT: Long press detected!");
62 }
63
64
65 // --- Check for State ---
66 // State checks return the current condition of the button.
67
68 // State: Currently being held down.
69 // This code block runs only every 200ms while the button is held
70 // to prevent flooding the serial monitor.
71 static unsigned long lastPrintTime = 0;
72 if (button.isPressed()) {
73 if (millis() - lastPrintTime > 200) {
74 Serial.println("STATE: Button is being held down...");
75 lastPrintTime = millis();
76 }
77 }
