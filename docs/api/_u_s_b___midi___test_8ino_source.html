45void setup() {
46 // Configure board for USB Host Mode
47 pinMode(PIN_USB_PWR_EN, OUTPUT);
48 digitalWrite(PIN_USB_PWR_EN, HIGH);
49 pinMode(PIN_FSUSB42MUX_SEL, OUTPUT);
50 digitalWrite(PIN_FSUSB42MUX_SEL, HIGH);
51 delay(200);
52
53 // Initialize USB Host Stack and our MIDI library
54 USBHost.begin(0);
55 midi.begin();
56
57 // Register callbacks
58 midi.setHandleNoteOn(handleNoteOn);
59 midi.setHandleNoteOff(handleNoteOff);
