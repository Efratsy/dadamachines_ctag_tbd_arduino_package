36/**
37 * @brief Audio Task that get
38 */
39void audioTask(void *pvParameters) {
40 delay(125);
41
42 // --- bring up the codec ---
43 if (!codec.begin(PIN_WIRE1_SDA, PIN_WIRE1_SCL)) {
44 Serial.println("Codec initialization failed! Halting.");
45 while (1);
46 }
47 codec.setHeadphoneVolume(70);
48 Serial.println("Codec initialized successfully.");
49
50 // --- configure the I2S engine (no loop yet) ---
51 CTAG_AudioEngine::init(I2S_NUM_0);
52
53 // --- hook up your VCO and defaults ---
54 CTAG_AudioEngine::setSource(&myVCO);
55 myVCO.setFrequency(220.0f);
56 myVCO.setAmplitude(0.5f);
57
58 Serial.println("Starting Audio Engine...");
59
60 // loop pumping audio ---
61 for (;;) {
62 CTAG_AudioEngine::renderBlock();
63
64 float freq = 220.0f + 220.0f * sin( millis() * 0.001f );
65 myVCO.setFrequency(freq);
66 }
67}
