20/**
21 * @brief Data structure for the UI state received from the extension board via I2C.
22 * @note This struct must exactly match the one on the slave device to ensure correct data parsing.
23 */
24typedef struct {
25 uint16_t pot_adc_values[8]; ///< RAW ADC values of the potentiometers.
26 uint16_t pot_positions[4]; ///< Absolute position of the potentiometers (0-65535).
27 uint8_t pot_states[4]; ///< State of the potentiometers (e.g., movement flags).
28 uint16_t d_btns; ///< Digital buttons D1-D16 (short press state, bitmask).
29 uint16_t d_btns_long_press; ///< Digital buttons D1-D16 (long press state, bitmask).
30 uint8_t f_btns; ///< Function buttons F1-F4 (short press state, bitmask).
31 uint8_t f_btns_long_press; ///< Function buttons F1-F4 (long press state, bitmask).
32 uint8_t encoder_state; ///< State of the encoder (button press, direction flags).
33 uint32_t encoder_counter; ///< 32-bit incremental counter of the encoder.
34 uint8_t encoder_absolute_pos; ///< Absolute position of the encoder.
35 uint8_t encoder_speed; ///< Rotational speed of the encoder.
36 uint32_t systicks; ///< Timestamp from the slave device.
37} ui_data_t;

46 /**

51 * @param wire A pointer to the TwoWire object to use (default: &Wire1 for RP2040).

53 CTAG_ExtensionBoard(uint8_t num_leds, uint8_t led_pin, uint8_t i2c_addr = 0x42, TwoWire* wire = &Wire1);
54
55 /**
56 * @brief Initializes the I2C connection and the LEDs.

59 * @param sclPin The pin for the I2C clock line (SCL).
60 */
61 void begin(int8_t sdaPin, int8_t sclPin);
62
63 /**
64 * @brief Retrieves the latest data from the extension board.

68 bool update();
