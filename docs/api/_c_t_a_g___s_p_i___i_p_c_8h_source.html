1/**
2 * @file CTAG_SPI_IPC.h
3 * @brief A simple, packet-based SPI library for inter-processor communication.
4 *
5 * @defgroup Libraries_SPI_IPC CTAG_SPI_IPC
6 * @ingroup Libraries
7 *
8 * This library provides a master/slave communication protocol over SPI,
9 * ensuring data integrity with a custom packet structure including magic bytes,
10 * length, and a CRC8 checksum. The master implementation is platform-independent,
11 * while the slave implementation is currently specific to the ESP32.
12 *
13 * 1. CTAG_SPI_IPC
14 */

24/**
25 * @brief Callback function type for the slave mode.
26 * @note On ESP32, this function is called from an interrupt service routine (ISR).
27 * Keep the code within the callback short and efficient.
28 */
29using Callback = void(*)(const uint8_t* data, size_t len);

31// --- Master Functions (Platform-Independent) ---
32
33/**
34 * @brief Initializes the module as a platform-independent SPI Master.
35 * @param csPin The Chip Select (CS) pin.
36 * @param spi A reference to the SPIClass object to use (default: SPI).
37 * @param speed The desired SPI clock speed in Hz (default: 8MHz).
38 * @param mode The SPI mode (default: SPI_MODE3).
39 * @return True on success.
40 */
41bool beginMaster(uint8_t csPin,
42 SPIClass& spi = SPI,
43 uint32_t speed = 8000000,
44 uint8_t mode = SPI_MODE3);
45
46/**
47 * @brief Sends a data packet as the SPI Master.
48 * @note The maximum payload size is 61 bytes due to the packet overhead.
49 * @param data Pointer to the data buffer to send.
50 * @param len The length of the data in bytes.
51 * @return True if the packet was sent, false on error (e.g., payload too large).
52 */
53bool send(const uint8_t* data, size_t len);
54
55
56// --- Slave Functions (ESP32 Specific Implementation) ---
57
58/**
59 * @brief Initializes the module as an SPI Slave (ESP32 only).
60 * @param sckPin The SPI Clock pin.
61 * @param misoPin The SPI MISO pin (Slave Out).
62 * @param mosiPin The SPI MOSI pin (Slave In).
63 * @param csPin The SPI Chip Select pin.
64 * @param cb Callback function that will be invoked when a new, valid packet arrives.
65 * @return True on success, false otherwise.
66 */
67bool beginSlave(uint8_t sckPin, uint8_t misoPin, uint8_t mosiPin, uint8_t csPin, Callback cb);
68
69/**
70 * @brief Checks if a new packet is available to be read in slave mode (polling method).
71 * @return True if a new packet has been received and validated.
72 */
73bool available();
74
75/**
76 * @brief Reads the last received packet into a user-provided buffer.
77 * @param buffer Pointer to the buffer where the data will be copied.
78 * @param maxLen The maximum size of the buffer.
79 * @return The number of bytes received, or 0 if no new packet was available.
80 */
81size_t receive(uint8_t* buffer, size_t maxLen);
