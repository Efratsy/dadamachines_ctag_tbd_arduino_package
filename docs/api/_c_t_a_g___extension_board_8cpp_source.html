28bool CTAG_ExtensionBoard::update() {
29 // Request the UI data structure from the I2C slave
30 _wire->requestFrom(_i2cAddr, sizeof(ui_data_t));
31
32 // Check if the correct number of bytes was received
33 if (_wire->available() < (int)sizeof(ui_data_t)) {
34 return false; // Not enough data received
35 }
36
37 // Read the data into the local buffer
38 _wire->readBytes((uint8_t*)&_data, sizeof(ui_data_t));
39 return true;
40}
