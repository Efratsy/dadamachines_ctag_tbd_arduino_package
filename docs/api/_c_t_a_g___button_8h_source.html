23class CTAG_Button {
24public:
25 /**
26 * @brief Constructs a new CTAG_Button object.
27 * @param pin The GPIO pin the button is connected to.
28 * @param debounceDelay The time in milliseconds to wait for the button signal to stabilize (default: 50).
29 * @param longPressDelay The time in milliseconds a button must be held down to be considered a long press (default: 800).

33 /**
34 * @brief Initializes the button's pin mode.

37 */
38 void begin(uint8_t mode = INPUT_PULLUP);
39
40 /**
41 * @brief Updates the button's internal state machine.
42 * @note This method must be called in every cycle of the main loop() to ensure proper operation.
43 */
44 void update();
45
46 /**
47 * @brief Checks if the button is currently being held down.
48 * @return True if the button is currently pressed, false otherwise.
49 */
50 bool isPressed();
51
52 /**
53 * @brief Checks if the button was just released after a short press.
54 * @note This is a "one-shot" event. It returns true for only one update() cycle after the event occurs.

57 bool wasShortPressed();
58
59 /**

63 */
64 bool wasLongPressed();
65
66private:
