14/**
15 * @brief Calculates the CRC8 checksum for a given data buffer.
16 * @param data Pointer to the data.
17 * @param len Length of the data.
18 * @return The calculated 8-bit CRC.
19 */
20static uint8_t calcCRC(const uint8_t* data, size_t len) {
21 uint8_t crc = 0;
22 while (len--) {
23 crc ^= *data++;
24 for (int i = 0; i < 8; ++i)
25 crc = (crc & 0x80) ? (crc << 1) ^ 0x07 : (crc << 1);
26 }
27 return crc;
28}

42bool send(const uint8_t* data, size_t len) {
43 // Check for initialization and valid length
44 if (!_spi || len > 61) return false;
45
46 // Construct the frame: [Magic1, Magic2, Len, Payload..., CRC]
47 uint8_t frame[64];
48 frame[0] = 0xCA;
49 frame[1] = 0xFE;
50 frame[2] = (uint8_t)len;
51 memcpy(&frame[3], data, len);
52 frame[3 + len] = calcCRC(&frame[3], len);
53 size_t frameLen = 3 + len + 1;
54
55 // Perform SPI transaction
56 _spi->beginTransaction(_settings);
57 digitalWrite(_csPin, LOW);
58 _spi->transfer(frame, frameLen);
59 digitalWrite(_csPin, HIGH);
60 _spi->endTransaction();
61
62 return true;
63}

80/**
81 * @brief ISR callback triggered after an SPI slave transaction is complete.
82 * @note This runs in an interrupt context. It validates the received packet
83 * and calls the user-provided callback if the packet is valid.
84 */
85static void IRAM_ATTR post_trans_cb(spi_slave_transaction_t *trans) {
86 // Check for magic bytes to identify a valid start of frame
87 if (_rxBuf[0] == 0xCA && _rxBuf[1] == 0xFE) {
88 uint8_t len = _rxBuf[2];
89 // Validate length and CRC checksum
90 if (len <= 61 && calcCRC(&_rxBuf[3], len) == _rxBuf[3 + len]) {
91 // If a user callback is registered, call it with the payload
92 if (_slaveCb) {
93 _slaveCb(&_rxBuf[3], len);
94 }
95 // Set the flag for the polling method (available()/receive())
96 _rxReady = true;
97 }
98 }
99 // Re-queue the transaction to be ready for the next one
100 spi_slave_queue_trans(SPI3_HOST, &_trans, portMAX_DELAY);
101}

103bool beginSlave(uint8_t sckPin, uint8_t misoPin, uint8_t mosiPin, uint8_t csPin, Callback cb) {
104 _slaveCb = cb;
105
106 // Configure the SPI bus
107 spi_bus_config_t buscfg = {
108 .mosi_io_num = mosiPin,
109 .miso_io_num = misoPin,
110 .sclk_io_num = sckPin,
111 .quadwp_io_num = -1,
112 .quadhd_io_num = -1,
113 .max_transfer_sz = BUFFER_SIZE
114 };
115
116 // Configure the SPI slave interface
117 spi_slave_interface_config_t slvcfg = {
118 .spics_io_num = csPin,
119 .flags = 0,
120 .queue_size = 1,
121 .mode = 0, // Mode 0 for ESP32 slave. Master must match.
122 .post_trans_cb = post_trans_cb
123 };
124
125 // Initialize the SPI slave driver
126 esp_err_t ret = spi_slave_initialize(SPI3_HOST, &buscfg, &slvcfg, SPI_DMA_CH_AUTO);
127 if (ret != ESP_OK) {
128 return false;
129 }
130
131 // Set up the transaction buffer details
132 memset(&_trans, 0, sizeof(_trans));
133 _trans.length = BUFFER_SIZE * 8; // Length in bits
134 _trans.rx_buffer = _rxBuf;
135 _trans.tx_buffer = _txBuf;
136
137 // Queue the first transaction
138 spi_slave_queue_trans(SPI3_HOST, &_trans, portMAX_DELAY);
139
140 return true;
141}
