2 * @file SPI_IPC_Slave.ino
3 * @brief A demonstration sketch for the CTAG_SPI_IPC library in Slave mode on an ESP32.

8 *
9 * The SPI_IPC_Slave.ino sketch configures an ESP32 as an SPI slave device that listens for packets
10 * from an SPI master. It uses a callback function to process incoming data
11 * asynchronously as soon as it arrives.
12 */
13#include <CTAG_SPI_IPC.h>
14#include "pins_arduino.h" // Required to use standardized pin names
15
16// --- Pin Definitions ---
17// Use the standardized pin names from your ESP32's "pins_arduino.h" file.
18// Note the crossover wiring required for SPI communication.
19#define PIN_MISO PIN_SPI0_MISO // Connects to Master's MOSI
20#define PIN_MOSI PIN_SPI0_MOSI // Connects to Master's MISO
21#define PIN_SCK PIN_SPI0_SCK
22#define PIN_CS PIN_SPI0_SS
23
24/**
25 * @brief Callback function that is automatically invoked when a valid SPI packet is received.
26 * @note This function is called from an interrupt context. Keep the code inside short
27 * and fast. Avoid using delay() or other long-running operations here.
28 * @param data A pointer to the received payload data.
29 * @param len The length of the payload data in bytes.
30 */
31void onPacketReceived(const uint8_t* data, size_t len) {
32 Serial.print(">> Packet Received! Length=");
33 Serial.print(len);
34 Serial.print(", Content: \"");
35 for (size_t i = 0; i < len; i++) {
36 Serial.print((char)data[i]);
37 }
38 Serial.println("\"");
39}

41/**
42 * @brief Runs once at startup to initialize the SPI Slave.
43 */
44void setup() {
45 Serial.begin(115200);
46 while(!Serial); // Wait for the serial port to connect
47 Serial.println("\n--- SPI Slave IPC Demo ---");
48
49 // Calling SPI.begin() is not necessary for the ESP32 slave mode,
50 // as the library passes the pins directly to the underlying ESP-IDF driver.
51
52 // Initialize the IPC library in slave mode, providing all necessary pins
53 // and the function to call when a packet arrives.
54 if (CTAG_SPI_IPC::beginSlave(PIN_SCK, PIN_MISO, PIN_MOSI, PIN_CS, onPacketReceived)) {
55 Serial.println("Slave ready and waiting for packets...");
56 } else {
57 Serial.println("Slave initialization FAILED!");
58 }
59}
