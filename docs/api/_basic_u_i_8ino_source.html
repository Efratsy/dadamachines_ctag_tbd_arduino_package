58/**
59 * @brief Runs continuously after setup to handle I/O.
60 */
61void loop() {
62 // --- 1. Always fetch the latest data from the board first. ---
63 if (!ext.update()) {
64 Serial.println("I2C connection failed!");
65 delay(200);
66 return;
67 }
68
69 // --- 2. Handle the LED logic (runs every loop for maximum responsiveness). ---
70
71 // Logic for the encoder LED (at index 0)
72 if (ext.isEncoderPressed()) {
73 ext.setLed(0, 0, 255, 0); // Green
74 } else if (ext.isEncoderRotatedForward() || ext.isEncoderRotatedBackward()) {
75 ext.setLed(0, 255, 0, 0); // Red on any rotation
76 } else {
77 ext.setLed(0, 0, 0, 0); // Off
78 }
79
80 // Logic for the 16 digital button LEDs (at indices 1-16)
81 for (int i = 0; i < 16; i++) {
82 uint8_t ledIndex = i + 1;
83 if (ext.isButtonLongPressed(i)) {
84 ext.setLed(ledIndex, 255, 0, 0); // Red for long press
85 } else if (ext.isButtonPressed(i)) {
86 ext.setLed(ledIndex, 0, 255, 0); // Green for short press
87 } else {
88 ext.setLed(ledIndex, 64, 64, 64); // Gray when idle
89 }
90 }
91
92 // --- 3. Send all updated LED colors to the hardware at once. ---
93 ext.show();
94
95
96 // --- 4. Handle serial output only every 'printInterval' milliseconds. ---
97 if (millis() - lastPrintTime >= printInterval) {
98 lastPrintTime = millis(); // Update the timestamp for the next print
99
100 // Output for the 4 potentiometers
101 Serial.print("Pots [0-3]:\t");
102 for (int i = 0; i < 4; i++) {
103 Serial.print(ext.getPot(i));
104 if (i < 3) {
105 Serial.print("\t"); // Tab for clean spacing
106 }
107 }
108 Serial.println(); // Newline for the next output
109
110 // Output for the 4 function buttons
111 Serial.print("F-Btns [0-3]:\t");
112 for (int i = 0; i < 4; i++) {
113 if (ext.isFButtonLongPressed(i)) {
114 Serial.print("LONG ");
115 } else if (ext.isFButtonPressed(i)) {
116 Serial.print("SHORT");
117 } else {
118 Serial.print(" OFF "); // Add spacing for consistent width
119 }
120
121 if (i < 3) {
122 Serial.print("\t"); // Tab for clean spacing
123 }
124 }
125 Serial.println(); // Final newline
126 Serial.println("------------------------------------"); // Separator for readability
127 }
