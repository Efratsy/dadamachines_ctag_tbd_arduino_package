121/**
122 * @brief Runs continuously after setup.
123 */
124void loop() {
125 // ALWAYS FIRST: Process any incoming MIDI data.
126 midi.read();
127
128 // --- Automatic Send Test ---
129 // Every `sendInterval` milliseconds, the next test message is sent.
130 if (millis() - lastSendTime > sendInterval) {
131 lastSendTime = millis();
132
133 Serial.println("------------------------------------------");
134
135 switch (sendState) {
136 case 0:
137 Serial.println("-> OUT: Sending Note On/Off (C4)...");
138 midi.sendNoteOn(60, 100, 1); // Note C4, Velocity 100, Channel 1
139 // This small delay is just to make the note audible in a loopback test.
140 // In real applications, avoid using delay() in the main loop.
141 delay(500);
142 midi.sendNoteOff(60, 0, 1);
143 break;
144
145 case 1:
146 { // Use a block to create a local scope for the variable.
147 byte ccValue = (millis() / 50) % 128; // A changing value.
148 Serial.print("-> OUT: Sending Control Change (CC 74, Value ");
149 Serial.print(ccValue);
150 Serial.println(")...");
151 midi.sendControlChange(74, ccValue, 1); // CC 74 (Filter Cutoff), Channel 1
152 }
153 break;
154
155 case 2:
156 {
157 byte program = random(1, 40);
158 Serial.print("-> OUT: Sending Program Change (Program ");
159 Serial.print(program);
160 Serial.println(")...");
161 midi.sendProgramChange(program, 1);
162 }
163 break;
164
165 case 3:
166 Serial.println("-> OUT: Sending Pitch Bend (UP)...");
167 midi.sendPitchBend(8191, 1); // Maximum pitch bend up.
168 break;
169
170 case 4:
171 Serial.println("-> OUT: Sending Pitch Bend (CENTER)...");
172 midi.sendPitchBend(0, 1); // Return pitch bend to center.
173 break;
174 }
175
176 sendState = (sendState + 1) % 5; // Advance to the next send state.
177 }
178}
