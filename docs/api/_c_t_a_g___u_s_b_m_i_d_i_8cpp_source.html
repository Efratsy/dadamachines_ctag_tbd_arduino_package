43void CTAG_USBMIDI::readPacket(uint8_t dev_addr) {
44 if (dev_addr != _dev_addr) return;
45
46 uint8_t packet[4];
47 while (tuh_midi_packet_read(dev_addr, packet)) {
48 // The first byte contains the Code Index Number (CIN), which tells us the message type.
49 uint8_t cin = packet[0] & 0x0F;
50 if (cin > 1) { // Ignore system messages for now, but parse the MIDI bytes
51 parse(packet[1]);
52 if(packet[2] != 0) parse(packet[2]);
53 if(packet[3] != 0) parse(packet[3]);
54 }
55 }
56}

96void CTAG_USBMIDI::sendPitchBend(int value, byte channel) {
97 if (_dev_addr == 0 || !tuh_midi_configured(_dev_addr)) return;
98 unsigned int midiValue = value + 8192;
99 byte lsb = midiValue & 0x7F;
100 byte msb = (midiValue >> 7) & 0x7F;
101 uint8_t packet[4] = { 0x0E, (uint8_t)(0xE0 | (channel - 1)), lsb, msb };
102 tuh_midi_packet_write(_dev_addr, packet);
103 tuh_midi_stream_flush(_dev_addr);
104}

126// --- PRIVATE PARSER ---
127void CTAG_USBMIDI::parse(byte midiByte) {
128 if (midiByte >= 0xF8) { return; } // Ignore Real-Time messages
129
130 if (midiByte & 0x80) { // It's a new status byte
131 status = midiByte;
132 data1 = 0; data2 = 0;
133 } else { // It's a data byte
134 if (data1 == 0) {
135 data1 = midiByte;
136 byte command = status & 0xF0;
137 if (command == 0xC0 || command == 0xD0) { // 2-byte message
138 if (handleProgramChange && command == 0xC0) { handleProgramChange((status & 0x0F) + 1, data1); }
139 status = 0; data1 = 0;
140 }
141 } else {
142 data2 = midiByte;
143 byte command = status & 0xF0;
144 byte channel = (status & 0x0F) + 1;
145 switch(command) {
146 case 0x90: // Note On
147 if (data2 > 0) { if (handleNoteOn) handleNoteOn(channel, data1, data2); }
148 else { if (handleNoteOff) handleNoteOff(channel, data1, data2); }
149 break;
150 case 0x80: // Note Off
151 if (handleNoteOff) handleNoteOff(channel, data1, data2);
152 break;
153 case 0xB0: // Control Change
154 if (handleControlChange) handleControlChange(channel, data1, data2);
155 break;
156 case 0xE0: // Pitch Bend
157 if (handlePitchBend) {
158 int value = ((int)data2 << 7) | (int)data1;
159 handlePitchBend(channel, value - 8192);
160 }
161 break;
162 }
163 data1 = 0; data2 = 0;
164 }
165 }
