88 static void audio_task(void* /*params*/) {
89 const int buffer_samples = 256;
90 int16_t i2s_buffer[buffer_samples * 2];
91 size_t bytes_written = 0;
92 while (true) {
93 for (int i = 0; i < buffer_samples * 2; i += 2) {
94 int16_t s = currentSource ? currentSource->getNextSample() : 0;
95 i2s_buffer[i] = s;
96 i2s_buffer[i + 1] = s;
97 }
98 i2s_write(_i2s_port,
99 i2s_buffer,
100 sizeof(i2s_buffer),
101 &bytes_written,
102 portMAX_DELAY);
103 }
104 }

106 void init(i2s_port_t i2s_port) {
107 _i2s_port = i2s_port;
108
109 // zero-init + configure
110 i2s_config_t cfg{};
111 cfg.mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_TX);
112 cfg.sample_rate = SAMPLE_RATE;
113 cfg.bits_per_sample = BITS_PER_SAMPLE;
114 cfg.channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT;
115 cfg.communication_format = I2S_COMM_FORMAT_STAND_I2S;
116 cfg.intr_alloc_flags = ESP_INTR_FLAG_LEVEL1;
117 cfg.dma_buf_count = 8;
118 cfg.dma_buf_len = 256;
119 cfg.use_apll = true;
120 cfg.tx_desc_auto_clear = true;
121 i2s_driver_install(_i2s_port, &cfg, 0, nullptr);
122
123 i2s_pin_config_t pin_cfg{};
124 pin_cfg.mck_io_num = PIN_I2S_MCLK;
125 pin_cfg.bck_io_num = PIN_I2S_BCLK;
126 pin_cfg.ws_io_num = PIN_I2S_WS;
127 pin_cfg.data_out_num = PIN_I2S_SDOUT;
128 pin_cfg.data_in_num = I2S_PIN_NO_CHANGE;
129 i2s_set_pin(_i2s_port, &pin_cfg);
130
131 i2s_set_clk(_i2s_port, SAMPLE_RATE, BITS_PER_SAMPLE, I2S_CHANNEL_STEREO);
132 }

143 void renderBlock() {
144 const int buffer_samples = 256;
145 int16_t buf[buffer_samples * 2];
146 size_t written;
147 for (int i = 0; i < buffer_samples; ++i) {
148 int16_t s = currentSource ? currentSource->getNextSample() : 0;
149 buf[2*i ] = s;
150 buf[2*i+1] = s;
151 }
152 i2s_write(_i2s_port, buf, sizeof(buf), &written, portMAX_DELAY);
153 }

199int16_t CTAG_VCO_Sine::getNextSample() {
200 // Advance LFO
201 _lfoPhase += _lfoIncrement;
202 if (_lfoPhase >= 2.0f * M_PI) _lfoPhase -= 2.0f * M_PI;
203 float vibrato = sin(_lfoPhase) * _lfoDepth;
204
205 // Instantaneous phase increment with vibrato
206 float instIncrement = (2.0f * M_PI * (_frequency + vibrato)) / _sampleRate;
207 _phase += instIncrement;
208 if (_phase >= 2.0f * M_PI) _phase -= 2.0f * M_PI;
209
210 float out = sin(_phase) * _amplitude;
211 return (int16_t)(out * 32767.0f);
212}

244 // Advance phase
245 _phase += _phaseIncrement;
246 if (_phase >= 2.0f * M_PI) {
247 _phase -= 2.0f * M_PI;
248 }
249
250 // Output high for the first portion of the cycle,
251 // then low for the remainder, scaled by amplitude.
252 float out = (_phase < (2.0f * M_PI * _dutyCycle) ? 1.0f : -1.0f)

286int16_t CTAG_VCO_Saw::getNextSample() {
287 // Advance phase
288 _phase += _phaseIncrement;
289 if (_phase >= 2.0f * M_PI) {
290 _phase -= 2.0f * M_PI;
291 }
292
293 // Normalize phase into [01)
294 float norm = _phase / (2.0f * M_PI);
295 float out;
296
297 // Rising ramp until skew, then falling ramp
298 if (norm < _skew) {
299 out = -1.0f + (norm / _skew) * 2.0f;
300 } else {
301 out = 1.0f - ((norm - _skew) / (1.0f - _skew)) * 2.0f;
302 }
303
304 // Apply amplitude
305 out *= _amplitude;
306
307 // Convert to 16-bit
308 return (int16_t)(out * 32767.0f);
309}
